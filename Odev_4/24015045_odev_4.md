```python
import cv2

cam=cv2.VideoCapture(0)
print(cam.get(3))
print(cam.get(4))
print(cam.get(5))


while True:
    ret,frame=cam.read()
    if not ret:
        print("goruntu alinamadi!!!")
        break
    
    frame=cv2.flip(frame,1)
    frame=cv2.cvtColor(frame,cv2.COLOR_BGR2GRAY)
    cv2.imshow("webcam",frame) #kamera oku
    k=cv2.waitKey(10)

    if  k==ord("q"):
        print("komut alindi.")
        break

cam.release()
cv2.destroyAllWindows()
```
