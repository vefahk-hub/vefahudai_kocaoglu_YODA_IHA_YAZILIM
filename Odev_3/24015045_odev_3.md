class kullanici: #kullanıcı sınıfı oluşturduk  
    def __init__(self, ad, soyad): #self kullanıcıların yerine geçecek diğerleri de kullanıcıların özellikleri olacak  
        self.ad = ad  
        self.soyad = soyad

        print(f"ad:{self.ad}, soyad{self.soyad}.")
